```bash
git config --global user.name ‘xxx’
git config --global user.email ‘xxx’
git config --list 查看配置
git init 初始化git仓库
git status 查看git状态
git add . 将代码提交到暂存区

# 此处的描述信息必须写，如果没写会进入一个黑窗口，你可以先按i键，然后写一段描述信息，写完以后按esc键，然后输入:wq
git commit -m '描述信息'
git checkout -- 文件名   丢弃工作区的修改
git reset HEAD 文件名    丢弃暂存区的修改
git commit -a -m '描述信息'  直接从工作区将代码提交到仓库，跳过中间的暂存区

git rm -f 文件名  删除文件，将删除的操作放在暂存区
git rm --cached 文件名  将文件移到工作区（其实就是删除了这个文件，然后创建了一个一模一样的文件）

git log 查看详细历史记录
git reflog 查看简洁历史记录

git reset --hard 版本ID  回退到指定版本
```
